CREATE TABLE EMPLOYEE_INFO
(
EMP_ID VARCHAR2(10)
CONSTRAINT EMPLOYEE_EMP_ID_PK PRIMARY KEY,
EMP_NAME FULLNAME
CONSTRAINT EMPLOYEE_EMP_NAME_NN NOT NULL,
DATE_OF_BIRTH DATE
CONSTRAINT EMPLOYEE_DOB_NN NOT NULL,
CONTACT_NUMBER NUMBER(10)
CONSTRAINT EMPLOYEE_CONTACT_NN NOT NULL,
EMAIL_ID VARCHAR2(50)
CONSTRAINT EMPLOYEE_MAIL_NN NOT NULL,
PRESENT_ADDRESS ADDRESS
CONSTRAINT EMPLOYEE_ADDRESS_NN NOT NULL,
DATE_OF_JOINING DATE
DEFAULT SYSDATE
CONSTRAINT EMPLOYEE_JOINDATE_NN NOT NULL,
DEPT_ID NUMBER(2)
CONSTRAINT EMPLOYEE_DEPT_ID_NN NOT NULL,
EMP_DESIGNATION VARCHAR2(20)
CONSTRAINT EMPLOYEE_DESG_NN NOT NULL,
CURRENT_PROJECT VARCHAR2(20)
DEFAULT 'NOT GIVEN'
CONSTRAINT EMPLOYEE_CURRENT_PROJECT_NN NOT NULL,
WORK_LOCATION VARCHAR2(20)
CONSTRAINT EMPLOYEE_LOCATION_NN NOT NULL
CHECK(
WORK_LOCATION IN
	(
	'HYDERABAD',
	'BANGALORE',
	'CHENNAI'
	)
),
CONSTRAINT EMPLOYEE_DEPT_ID_FK FOREIGN KEY(DEPT_ID)
REFERENCES DEPARTMENT_DETAILS(DEPT_ID)
ON DELETE CASCADE
);
